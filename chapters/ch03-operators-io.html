🔹 ตัวดำเนินการทางคณิตศาสตร์ (Arithmetic Operators)

ตัวดำเนินการทางคณิตศาสตร์ใช้สำหรับการคำนวณตัวเลข Python รองรับการดำเนินการพื้นฐานครบทุกรูปแบบ ดังตาราง:

ตัวดำเนินการ ชื่อ ตัวอย่าง ผลลัพธ์
+บวก (Addition)10 + 313
-ลบ (Subtraction)10 - 37
*คูณ (Multiplication)10 * 330
/หาร (Division)10 / 33.3333...
//หารเอาจำนวนเต็ม (Floor Division)10 // 33
%หารเอาเศษ (Modulo)10 % 31
**ยกกำลัง (Exponentiation)10 ** 31000
Python 
# ตัวดำเนินการทางคณิตศาสตร์
a, b = 10, 3

print(f"{a} + {b} = {a + b}")    # 13
print(f"{a} - {b} = {a - b}")    # 7
print(f"{a} * {b} = {a * b}")    # 30
print(f"{a} / {b} = {a / b:.4f}")  # 3.3333
print(f"{a} // {b} = {a // b}")  # 3  (หารเอาจำนวนเต็ม)
print(f"{a} % {b} = {a % b}")    # 1  (เศษ)
print(f"{a} ** {b} = {a ** b}")  # 1000 (ยกกำลัง)

# ตัวอย่างการใช้งานจริง
# หาว่า 100 หารด้วย 7 ได้กี่ครั้ง และเหลือเศษเท่าไร
dividend = 100
divisor = 7
quotient = dividend // divisor
remainder = dividend % divisor
print(f"{dividend} ÷ {divisor} = {quotient} เหลือเศษ {remainder}")
💡 การใช้งาน Modulo (%)
Modulo มีประโยชน์มากในการเขียนโปรแกรม เช่น ตรวจสอบว่าตัวเลขเป็นเลขคู่หรือเลขคี่ (n % 2 == 0 หมายถึงเลขคู่) หรือหาว่าตัวเลขหารลงตัวหรือไม่
🔹 ตัวดำเนินการเปรียบเทียบ (Comparison Operators)

ตัวดำเนินการเปรียบเทียบใช้เปรียบเทียบค่าสองค่า และให้ผลลัพธ์เป็น bool (True หรือ False) เสมอ

ตัวดำเนินการ ความหมาย ตัวอย่าง (a=5, b=3) ผลลัพธ์
==เท่ากับ (Equal to)a == bFalse
!=ไม่เท่ากับ (Not equal to)a != bTrue
<น้อยกว่า (Less than)a < bFalse
>มากกว่า (Greater than)a > bTrue
<=น้อยกว่าหรือเท่ากับa <= bFalse
>=มากกว่าหรือเท่ากับa >= bTrue
Python 
# ตัวดำเนินการเปรียบเทียบ
a, b = 5, 3

print(a == b)   # False
print(a != b)   # True
print(a > b)    # True
print(a < b)    # False
print(a >= 5)   # True
print(b <= 3)   # True

# Python รองรับการเปรียบเทียบต่อเนื่อง (Chained Comparison)
score = 75
print(60 <= score < 80)    # True (เกรด C)
print(0 <= score <= 100)   # True (คะแนนถูกต้อง)
🔹 ตัวดำเนินการตรรกะ (Logical Operators)

ตัวดำเนินการตรรกะ

ใช้รวมเงื่อนไขหลายๆ เงื่อนไขเข้าด้วยกัน มี 3 ตัว ได้แก่ and, or, not

  • and: จริงก็ต่อเมื่อทั้งสองเงื่อนไขเป็นจริง
  • or: จริงเมื่ออย่างน้อยหนึ่งเงื่อนไขเป็นจริง
  • not: กลับค่าความจริง (True → False, False → True)
Python 
# ตัวดำเนินการตรรกะ
age = 20
has_id = True
is_member = False

# and — ต้องจริงทั้งคู่
print(age >= 18 and has_id)     # True
print(age >= 18 and is_member)  # False

# or — จริงอย่างน้อยหนึ่งอย่าง
print(age >= 18 or is_member)   # True
print(age < 18 or is_member)    # False

# not — กลับค่า
print(not is_member)            # True
print(not has_id)               # False

# ตัวอย่างจริง: ตรวจสอบสิทธิ์เข้าใช้
username = "admin"
password = "1234"

if username == "admin" and password == "1234":
    print("เข้าสู่ระบบสำเร็จ")
else:
    print("ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง")
🔹 ตัวดำเนินการกำหนดค่า (Assignment Operators)
ตัวดำเนินการ ตัวอย่าง เทียบเท่ากับ
=x = 5กำหนดค่า 5 ให้ x
+=x += 3x = x + 3
-=x -= 3x = x - 3
*=x *= 3x = x * 3
/=x /= 3x = x / 3
//=x //= 3x = x // 3
%=x %= 3x = x % 3
**=x **= 3x = x ** 3
Python 
# ตัวดำเนินการกำหนดค่าแบบย่อ
score = 0
print(f"เริ่มต้น: {score}")

score += 10   # เพิ่ม 10
print(f"หลัง += 10: {score}")

score *= 2    # คูณ 2
print(f"หลัง *= 2: {score}")

score -= 5    # ลด 5
print(f"หลัง -= 5: {score}")

score //= 3   # หารเอาจำนวนเต็ม
print(f"หลัง //= 3: {score}")
🔹 ลำดับความสำคัญของตัวดำเนินการ (Operator Precedence)

ลำดับความสำคัญ (PEMDAS)

เมื่อมีตัวดำเนินการหลายตัวในนิพจน์เดียวกัน Python จะคำนวณตามลำดับความสำคัญ ดังนี้ (จากสูงไปต่ำ)

  1. ( ) — วงเล็บ (Parentheses) สูงสุด
  2. ** — ยกกำลัง (Exponentiation)
  3. +x, -x — เครื่องหมายบวก/ลบหน้าตัวเลข (Unary)
  4. *, /, //, % — คูณ หาร (Multiplication / Division)
  5. +, - — บวก ลบ (Addition / Subtraction)
  6. ==, !=, <, >, <=, >= — การเปรียบเทียบ
  7. not — ตรรกะ not
  8. and — ตรรกะ and
  9. or — ตรรกะ or (ต่ำสุด)
Python 
# ลำดับความสำคัญ
print(2 + 3 * 4)       # 14 (คูณก่อนบวก)
print((2 + 3) * 4)     # 20 (วงเล็บก่อน)
print(2 ** 3 ** 2)     # 512 (** คำนวณจากขวาไปซ้าย: 3**2=9, 2**9=512)
print(10 - 4 / 2)      # 8.0 (หารก่อน: 4/2=2.0, 10-2=8.0)
print(10 - (4 / 2))    # 8.0 (ใช้วงเล็บให้ชัดเจน ผลเหมือนกัน)

# แนะนำให้ใช้วงเล็บเสมอเพื่อความชัดเจน
a, b, c = 2, 3, 4
result = (a + b) * c - a ** 2
print(f"({a} + {b}) * {c} - {a}² = {result}")
🔹 การแสดงผล — print()

ฟังก์ชัน print()

ฟังก์ชัน print() ใช้แสดงผลลัพธ์ออกหน้าจอ มี parameter พิเศษที่มีประโยชน์ คือ

  • sep — ตัวคั่นระหว่างค่าต่างๆ (ค่าเริ่มต้นคือ " " เว้นวรรค)
  • end — ตัวอักษรที่ต่อท้าย (ค่าเริ่มต้นคือ "\n" ขึ้นบรรทัดใหม่)
Python 
# การใช้ print() แบบต่างๆ
print("A", "B", "C")               # A B C (คั่นด้วยเว้นวรรค)
print("A", "B", "C", sep="-")      # A-B-C (คั่นด้วย -)
print("A", "B", "C", sep="")       # ABC (ไม่มีตัวคั่น)
print("A", "B", "C", sep="\n")     # A B C คนละบรรทัด

print("บรรทัดที่ 1", end=" ")      # ไม่ขึ้นบรรทัดใหม่
print("บรรทัดที่ 2")               # ต่อท้ายบรรทัดเดิม

# พิมพ์ตัวเลข
print(42, 3.14, True)              # 42 3.14 True
🔹 การจัดรูปแบบข้อความ — f-string

f-string (Formatted String Literals)

f-string เป็นวิธีการจัดรูปแบบข้อความที่ทันสมัยและนิยมใช้ใน Python 3.6+ โดยขึ้นต้นสตริงด้วยตัวอักษร f หรือ F และใส่ตัวแปรหรือนิพจน์ไว้ในวงเล็บปีกกา { }

Python 
# f-string พื้นฐาน
name = "สมชาย"
age = 20
gpa = 3.75

print(f"ชื่อ: {name}")
print(f"อายุ: {age} ปี")
print(f"เกรดเฉลี่ย: {gpa}")

# นิพจน์ใน f-string
a, b = 15, 4
print(f"{a} + {b} = {a + b}")
print(f"{a} × {b} = {a * b}")

# จัดรูปแบบตัวเลข
pi = 3.14159265
print(f"π ≈ {pi:.2f}")        # 2 ตำแหน่งทศนิยม
print(f"π ≈ {pi:.5f}")        # 5 ตำแหน่งทศนิยม

salary = 35000
print(f"เงินเดือน: {salary:,} บาท")   # ใส่ comma คั่นหลักพัน

score = 85.5
print(f"คะแนน: {score:>8.1f}")  # จัดชิดขวา ความกว้าง 8

# วิธีเก่า (ยังใช้ได้ แต่ f-string แนะนำกว่า)
print("ชื่อ: %s, อายุ: %d" % (name, age))
print("เกรด: {:.2f}".format(gpa))
🔹 การรับข้อมูล — input()

ฟังก์ชัน input()

ฟังก์ชัน input() ใช้รับข้อมูลจากผู้ใช้ผ่านแป้นพิมพ์ โดยจะหยุดรอจนกว่าผู้ใช้จะกด Enter สำคัญมาก: input() คืนค่าเป็น str เสมอ ไม่ว่าผู้ใช้จะพิมพ์ตัวเลขหรือข้อความ ดังนั้นหากต้องการใช้ในการคำนวณต้องแปลงชนิดข้อมูลก่อน

Python 
# input() คืนค่าเป็น str เสมอ
# ตัวอย่างที่ผิด (จะเกิด TypeError ถ้าผู้ใช้กรอกตัวเลข)
# age = input("กรอกอายุ: ")
# print(age + 5)   # ERROR! ไม่สามารถบวก str กับ int ได้

# ตัวอย่างที่ถูกต้อง
# age = int(input("กรอกอายุ: "))
# print(age + 5)   # ถูกต้อง

# เนื่องจากในเบราว์เซอร์ input() ไม่สามารถรับ input จริงๆ ได้
# ในตัวอย่างนี้เราจำลองด้วยการกำหนดค่าตรงๆ
name = "สมหมาย"              # ปกติ: name = input("กรอกชื่อ: ")
age = 21                     # ปกติ: age = int(input("กรอกอายุ: "))
height = 1.68                # ปกติ: height = float(input("กรอกส่วนสูง (เมตร): "))

print(f"สวัสดี คุณ{name}!")
print(f"อายุ: {age} ปี")
print(f"ส่วนสูง: {height:.2f} เมตร")
print(f"BMI จะคำนวณได้เมื่อทราบน้ำหนักด้วย")

🔹 ตัวอย่างโปรแกรมจริง: คำนวณ BMI
Python 
# โปรแกรมคำนวณ BMI (Body Mass Index)
# BMI = น้ำหนัก(kg) / ส่วนสูง(m)²
# (จำลอง input ด้วยค่าตรงๆ เพื่อทดสอบ)

name = "สมชาย"
weight = 70.0    # กิโลกรัม  (int(input("น้ำหนัก(kg): ")))
height = 1.75    # เมตร      (float(input("ส่วนสูง(m): ")))

bmi = weight / (height ** 2)

print("=" * 35)
print(f"  ผลการคำนวณ BMI ของ: {name}")
print("=" * 35)
print(f"  น้ำหนัก  : {weight:.1f} kg")
print(f"  ส่วนสูง  : {height:.2f} m")
print(f"  BMI      : {bmi:.2f}")
print("=" * 35)

if bmi < 18.5:
    category = "น้ำหนักน้อยกว่าเกณฑ์"
elif bmi < 25.0:
    category = "น้ำหนักปกติ"
elif bmi < 30.0:
    category = "น้ำหนักเกิน"
else:
    category = "อ้วน"

print(f"  สถานะ    : {category}")

🔹 คำถามทบทวน

คำถามทบทวนบทที่ 3

  1. อธิบายความแตกต่างระหว่างตัวดำเนินการ / และ // ใน Python พร้อมยกตัวอย่างที่ผลลัพธ์แตกต่างกัน และบอกว่าแต่ละอย่างเหมาะกับสถานการณ์ใด
  2. ผลลัพธ์ของนิพจน์ต่อไปนี้คืออะไร? อธิบายตามลำดับความสำคัญของตัวดำเนินการ: 3 + 4 * 2 - 1, 2 ** 3 + 1, 10 % 3 + 1
  3. ทำไม input() จึงคืนค่าเป็น str เสมอ? จะเกิดอะไรขึ้นถ้าเราใช้ค่านั้นในการคำนวณโดยไม่แปลงชนิดข้อมูล? จงเขียนโค้ดแสดงปัญหาและวิธีแก้
  4. จงเขียนโปรแกรม Python ที่รับรัศมีของวงกลม แล้วคำนวณและแสดงผล: พื้นที่วงกลม (π r²) และเส้นรอบวง (2π r) โดยใช้ f-string จัดรูปแบบผลลัพธ์ให้แสดงทศนิยม 4 ตำแหน่ง
  5. อธิบายความแตกต่างระหว่าง and และ or พร้อมสร้างตาราง Truth Table สำหรับแต่ละตัว และยกตัวอย่างสถานการณ์จริงที่ใช้แต่ละตัว
