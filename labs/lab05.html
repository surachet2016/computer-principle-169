📋 ขั้นตอนที่ 1: for loop กับ range()

for loop ใช้วนซ้ำตามจำนวนครั้งที่กำหนด ฟังก์ชัน range() สร้างลำดับตัวเลขให้ โดย range(n) คือ 0 ถึง n-1, range(start, end) คือ start ถึง end-1 และ range(start, end, step) คือเพิ่มทีละ step ข้อสำคัญ: ค่าสุดท้ายของ range จะไม่รวมอยู่ใน loop

Python 
# range(n) — วนจาก 0 ถึง n-1
for i in range(5):
    print(i, end=" ")   # 0 1 2 3 4
print()

# range(start, end) — วนจาก start ถึง end-1
for i in range(1, 6):
    print(i, end=" ")   # 1 2 3 4 5
print()

# range(start, end, step) — กำหนดก้าว
for i in range(0, 20, 5):
    print(i, end=" ")   # 0 5 10 15
print()

# นับถอยหลัง (step ติดลบ)
for i in range(5, 0, -1):
    print(i, end=" ")   # 5 4 3 2 1
print()

📋 ขั้นตอนที่ 2: for loop กับ list และ string

for loop ใน Python วนซ้ำผ่านสิ่งที่เรียก iterable ได้ทุกชนิด เช่น list, string หรือ tuple โดยตัวแปรใน for จะรับค่าแต่ละตัวของ iterable ทีละตัวโดยอัตโนมัติ

Python 
# วนซ้ำผ่าน list
fruits = ["แอปเปิล", "กล้วย", "ส้ม", "มะม่วง"]
for fruit in fruits:
    print(f"ผลไม้: {fruit}")

print()

# วนซ้ำผ่าน string (ทีละตัวอักษร)
word = "Python"
for char in word:
    print(char, end="-")   # P-y-t-h-o-n-
print()

# ใช้ enumerate() เพื่อดัชนีด้วย
for i, fruit in enumerate(fruits, start=1):
    print(f"{i}. {fruit}")

📋 ขั้นตอนที่ 3: while loop

while loop วนซ้ำตราบเท่าที่เงื่อนไขยังเป็น True ใช้เมื่อไม่รู้ล่วงหน้าว่าจะวนกี่รอบ ต้องระวัง infinite loop (วนไม่สิ้นสุด) ให้มั่นใจว่าตัวแปรในเงื่อนไขมีการเปลี่ยนค่าในทุกรอบ ไม่เช่นนั้นโปรแกรมจะค้าง

Python 
# while loop พื้นฐาน
count = 1
while count <= 5:
    print(f"รอบที่ {count}")
    count += 1   # อย่าลืมเพิ่มค่า! ไม่งั้น loop ไม่สิ้นสุด

# นับถอยหลัง
n = 5
while n > 0:
    print(n, end=" ")
    n -= 1
print("Go!")

# หาผลรวม 1 ถึง 10
total = 0
i = 1
while i <= 10:
    total += i
    i += 1
print(f"ผลรวม 1-10 = {total}")

📋 ขั้นตอนที่ 4: break และ continue

break ใช้หยุด loop ทันทีและออกจาก loop นั้น ส่วน continue ใช้ข้ามรอบปัจจุบันและไปเริ่มรอบถัดไปโดยไม่ออกจาก loop ทั้งสองคำสั่งนิยมใช้ร่วมกับ if เพื่อควบคุมการทำงานของ loop อย่างละเอียด

Python 
# break — หยุด loop เมื่อเจอค่าที่ต้องการ
for i in range(1, 11):
    if i == 6:
        print(f"พบเลข {i} — หยุด!")
        break
    print(i, end=" ")
print()

# continue — ข้ามรอบที่ไม่ต้องการ
print("เลขคู่ 1-10:")
for i in range(1, 11):
    if i % 2 != 0:   # ถ้าเลขคี่
        continue      # ข้ามไป
    print(i, end=" ")
print()
